var files = "src/**.cpp"

var debug_flags = [
    "-Wall",
    "-Wextra",
    "-DDEBUG"
]

var release_flags = [
    "-O3",
    "-DNDEBUG"
]

var program_name = "tmake"
var output_directory = "./bin"

COMPILER("g++")
VERSION("17")

if "$1" == "release" {
    PROGRAM(program_name, files, release_flags, output_directory)
} else {
    PROGRAM(program_name, files, debug_flags, output_directory)
}
